import datetime

def elegir_fecha():
    fechas_posibles = [
        ("10/10", "Manurro"),
        ("21/09", "Matute"),
        ("22/11", "Marga"),
        ("25/04", "Apa")
    ]
    print("Elige una de las siguientes fechas:")
    for i, (fecha, nombre) in enumerate(fechas_posibles, start=1):
        print(f"{i}. {fecha} ({nombre})")
    
    opcion = int(input("Elige el número de la fecha correcta (1-4): "))
    if opcion == 2:
        print("¡Correcto! El regalo viene de Matute.")
    else:
        print("¡Incorrecto! Sigue buscando...")

elegir_fecha()
